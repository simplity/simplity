210 /**
211  * Simple way to get response from your service
212  */
213 var Simplity = (function() {
214 	/**
215 	 * simple function to escape "<" ad "&" so that innerHTML is safe
216 	 * 
217 	 * @method
218 	 * @param {string}
219 	 *            text - text to be escaped
220 	 */
221 	var htmlEscape = function(text) {
222 		if (!text || !text.replace) {
223 			return text;
224 		}
225 		return text.replace(/&/g, "&").replace(/</g, "<");
226 	};
227 
228 	/**
229 	 * @method log an error message.
230 	 * @param {string}
231 	 *            msg - to be logged
232 	 */
233 	var error = function(msg) {
234 		alert("ERROR\n" + msg);
235 	};
236 
237 	/**
238 	 * @method log a warning text
239 	 * @param {string}
240 	 *            msg - to be logged
241 	 */
242 	var warn = function(msg) {
243 		alert("Warning\n" + msg);
244 	};
245 
246 	/**
247 	 * @method. log a message
248 	 * @param {string}
249 	 *            msg - to be logged
250 	 * @param {boolean}
251 	 *            isHtml - if this is formatted as html. (In case log is written
252 	 *            out to as html, this msg is not escaped)
253 	 */
254 	var log = function(msg, isHtml) {
255 		console.log(msg);
256 	};
257 	/**
258 	 * @method pipe logging from our default console to your standard functions
259 	 * 
260 	 * @param {
261 	 *            function } errorFn - function for reporting error - required.
262 	 * @param {function}
263 	 *            warningFn - function for reporting warning - required.
264 	 * @param {function}
265 	 *            logFn - optional function for logging. If this is null, log
266 	 *            will continue to go to console
267 	 */
268 	var pipeLogging = function(errorFn, warnFn, logFn) {
269 		if (errorFn && warnFn) {
270 			error = errorFn;
271 			warn = warningFn;
272 		} else {
273 			reportError('You need to specify both errorFn and warningFn for re-plumbing logging with pipeLogging() method');
274 			return;
275 		}
276 		if (logFn) {
277 			log = logFn;
278 		}
279 	};
280 
281 	var showMessage = function(message) {
282 		alert(message);
283 	};
284 
285 	var showMessages = function(messages) {
286 		if (!messages || !messages.length) {
287 			alert('There are no messages');
288 		}
289 		var t = [];
290 		for (var i = 0; i < messages.length; i++) {
291 			var msg = messages[i];
292 			if (msg.messageType) {
293 				t.push(msg.messageType.toUpperCase());
294 				t.push('\n');
295 			}
296 			if(msg.name){
297 				t.push(msg.name);
298 				t.push(' : ');
299 			}
300 			t.push(msg.text);
301 			t.push('\n\n');
302 		}
303 		showMessage(t.join(''));
304 	};
305 
306 	/**
307 	 * @method over-ride our alert to any exotic way of showing message.
308 	 * @param {function}
309 	 *            your function that over-rides the default
310 	 */
311 	var overrideShowMessage = function(fn) {
312 		showMessage = fn;
313 	};
314 	/**
315 	 * @method over-ride our alert to any exotic way of showing message.
316 	 * @param {function}
317 	 *            your function that over-rides the default
318 	 */
319 	var overrideShowMessages = function(fn) {
320 		showMessages = fn;
321 	};
322 	/**
323 	 * agent who knows how to contact server
324 	 */
325 	var METHOD = "POST";
326 	var URL = 's';
327 	var LOGIN_URL = 'i';
328 	var LOGOUT_URL = 'o';
329 	var TIMEOUT = 120000;
330 	/**
331 	 * this is the token to be flashed at the server security guard. This is
332 	 * obtained after a login.
333 	 */
334 	var token = null;
335 	/**
336 	 * @method login to server with credentials
337 	 * @param {text}
338 	 *            login name
339 	 * @param {text}
340 	 *            abrakadabra
341 	 */
342 	var login = function(userId, userToken) {
343 		var xhr = new XMLHttpRequest();
344 		xhr.onreadystatechange = function() {
345 			if (this.readyState == '4') {
346 				token = xhr.getResponseHeader(POCOL.CSRF_HEADER);
347 				if (token) {
348 					showMessages([{messageType:'success', text:'Successfully logged-in.'}]);
349 				} else {
350 					showMessage([{messageType:'error', name:'InvalidCredentials',text:
351 							'Ooops. Your credentials were not honoured. Please re-try.'}]);
352 				}
353 			}
354 		};
355 		xhr.ontimeout = function() {
356 			showMessages([{messageType:"error", name:"serverTimeout", text:
357 					"Sorry, there seem to be some red-tapism on the server. Can't wait any more for a decision. Giving up."}]);
358 		};
359 		try {
360 			xhr.open(METHOD, LOGIN_URL, true);
361 			xhr.timeout = TIMEOUT;
362 			xhr.setRequestHeader("Content-Type", "text/html; charset=utf-8");
363 			xhr.setRequestHeader(POCOL.USER_ID, userId);
364 			if (userToken) {
365 				xhr.setRequestHeader(POCOL.USER_TOKEN, userToken);
366 			}
367 			xhr.send('');
368 		} catch (e) {
369 			showMessages([{messageType:'error', text:'Unable to connect to server. Error : ' + e}]);
370 		}
371 	};
372 
373 	/**
374 	 * @method one-way communication - we tell server to logout
375 	 */
376 	var logout = function() {
377 		var xhr = new XMLHttpRequest();
378 		try {
379 			xhr.open(METHOD, LOGOUT_URL, true);
380 			xhr.setRequestHeader("Content-Type", "text/html; charset=utf-8");
381 			if (token) {
382 				xhr.setRequestHeader(POCOL.USER_TOKEN, token);
383 			}
384 			xhr.send('');
385 		} catch (e) {
386 			showMessages([{text:'Unable to connect to server. Error : ' + e, messageType:'error'}]);
387 		}
388 		/*
389 		 * our token is no more valid
390 		 */
391 		token = null;
392 	};
393 
394 	/**
395 	 * returns a promise for the specified service that is invoked with supplied
396 	 * data
397 	 */
398 	var getResponse = function(serviceName, data) {
399 		return new Promise(function(fullfill, reject) {
400 			if (!serviceName) {
401 				log('No service');
402 				reject([{
403 						messageType : 'error',
404 					text : 'No serviceName specified'
405 				}]);
406 				return;
407 			}
408 			log('Service ' + serviceName + ' invoked');
409 			var xhr = new XMLHttpRequest();
410 			xhr.onreadystatechange = function() {
411 				if (this.readyState != '4') {
412 					return;
413 				}
414 				/*
415 				 * any issue with HTTP Connection?
416 				 */
417 				if (xhr.status != 200 && xhr.status != 0) {
418 					log('non 200 status : ' + xhr.status);
419 					reject([{
420 						messageType : 'error',
421 						text : 'Our courier returned with an error code='
422 								+ xhr.status
423 								+ ' look at logs for more details.'
424 					}]);
425 					return;
426 				}
427 				/*
428 				 * any issue with our web agent?
429 				 */
430 				var st = xhr.getResponseHeader(POCOL.REQUEST_STATUS);
431 				if(st == null){
432 					reject([{text:'Unknown error on the server. It failed to indicate any status.', messageType:'error'}]);
433 					return;
434 				}
435 				var ms = xhr.getResponseHeader(POCOL.SERVICE_EXECUTION_TIME);
436 				if(ms){
437 					log('server claims it took ' + ms +'ms');
438 				}
439 				var json = xhr.responseText ? JSON.parse(xhr.responseText) : {};
440 				if (st === POCOL.REQUEST_OK) {
441 					fullfill(json);
442 				} else {
443 					log('service returned with status=' + st);
444 					reject(json);
445 				}
446 			};
447 			xhr.ontimeout = function() {
448 				log('time out');
449 				reject([{messageType : 'error', text:'Sorry, there seem to be some red-tapism on the server. giving-up'}]);
450 			};
451 			try {
452 				xhr.open(METHOD, URL, true);
453 				xhr.timeout = TIMEOUT;
454 				xhr
455 						.setRequestHeader("Content-Type",
456 								"text/html; charset=utf-8");
457 				xhr.setRequestHeader(POCOL.SERVICE_NAME, serviceName);
458 				if (token) {
459 					xhr.setRequestHeader(POCOL.USER_TOKEN, token);
460 				}
461 				xhr.send(data);
462 			} catch (e) {
463 				error("error during xhr : " + e);
464 				reject([ {
465 					text : 'Unable to connect to server. Error : ' + e,
466 					messageType : 'error'
467 				} ]);
468 			}
469 
470 		});
471 	};
472 
473 	/*
474 	 * what we want to expose as API
475 	 */
476 	return {
477 		htmlEscape : htmlEscape,
478 		log : log,
479 		error : error,
480 		warn : warn,
481 		pipeLogging : pipeLogging,
482 		showMessage : showMessage,
483 		showMessages : showMessages,
484 		overrideShowMessage : overrideShowMessage,
485 		overrideShowMessages : overrideShowMessages,
486 		getResponse : getResponse,
487 		login : login,
488 		logout : logout,
489 	};
490 })();
